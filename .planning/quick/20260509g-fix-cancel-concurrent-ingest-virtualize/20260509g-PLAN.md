```rust
/// 跨 game query dedup cache — scope 限定为单次 start_scan 调用。
///
/// 多个并发 ingest task 命中同一 cleaned query 时，OnceCell::get_or_init
/// 保证只发一次 Bangumi+VNDB 搜索；其他 task 直接复用 Option<Candidate>。
///
/// 仅 start_scan 路径用；add_game / refresh_metadata / bind_metadata /
/// refresh_all_metadata 不传 cache，process_game / refresh_for_query
/// 公开签名因此完全不变。
pub type QueryCache = Mutex<HashMap<String, Arc<OnceCell<Option<Candidate>>>>>;

pub fn new_query_cache() -> Arc<QueryCache> {
    Arc::new(Mutex::new(HashMap::new()))
}

/// `pick_best_across_sources` 的 cache wrapper。命中 cache 复用结果；
/// 未命中通过 OnceCell::get_or_init 防 thundering-herd。
///
/// `Option<Candidate>` 是 already-merged + filtered 的最优 candidate，
/// 与 `pick_best_across_sources` 返回值语义一致。
pub async fn pick_best_with_cache(query: &str, cache: &QueryCache) -> Option<Candidate> {
    let cell = {
        let mut g = cache.lock().expect("query cache mutex poisoned");
        g.entry(query.to_string())
            .or_insert_with(|| Arc::new(OnceCell::new()))
            .clone()
    };
    cell.get_or_init(|| async move {
        pick_best_across_sources(query).await
    })
    .await
    .clone()
}
```

**注意 Candidate 已派生 Clone**（验证 ingest.rs use 处的 Candidate import 来自 metadata::Candidate；如果 Candidate 没派生 Clone，Task 2 不要修改 metadata/mod.rs，改为存 `Arc<Option<Candidate>>` 并 clone Arc — 但 metadata/mod.rs 当前确实派生了 Clone，先用 clone）。

主体：
```tsx
import { useCallback, useEffect, useMemo, useRef, useState } from "react";
import { convertFileSrc, invoke } from "@tauri-apps/api/core";
import { toast } from "sonner";
import { useVirtualizer } from "@tanstack/react-virtual";
import { GameCard } from "./GameCard";
import { refreshMetadata } from "@/lib/metadata";
import { listGames, type Game } from "@/lib/games";
import { useLibraryStore } from "@/store/library";
import { searchGames } from "@/lib/search";

interface GameGridProps {
  games: Game[];
  onPickMetadata: (game: Game) => void;
  onChildMutation?: () => void;
  /** 由 Library.tsx 持有的滚动容器 ref —— useVirtualizer 的 getScrollElement 目标。 */
  scrollContainerRef: React.RefObject<HTMLDivElement | null>;
}

const COLUMN_GAP = 22;  // 与下面 inline style 的 gap "28px 22px" 第二个值一致
const ROW_GAP = 28;     // 同上第一个值
const META_HEIGHT = 56; // cover 之外（meta + 卡片下方 gap）

export function GameGrid({
  games,
  onPickMetadata,
  onChildMutation,
  scrollContainerRef,
}: GameGridProps) {
  const setGames = useLibraryStore((s) => s.setGames);

  const [dataDir, setDataDir] = useState<string | null>(null);
  useEffect(() => {
    invoke<string>("get_data_dir")
      .then(setDataDir)
      .catch((e: unknown) => console.error("[GameGrid] get_data_dir failed:", e));
  }, []);

  const resolveCover = useMemo(() => {
    return (game: Game): string | null => {
      if (game.cover_path && dataDir) {
        const abs = `${dataDir.replace(/\\/g, "/")}/${game.cover_path}`;
        return convertFileSrc(abs);
      }
      return game.cover_url ?? null;
    };
  }, [dataDir]);

  // ── ResizeObserver-driven columnCount + cardWidth ─────────────────────────
  // 测量「内层容器」（GameGrid 的 root，不是 scrollContainer）的可用宽度，
  // 减去 px-8 内 padding（左右各 32px = 64px），除以 (cardWidth + COLUMN_GAP)
  // 推 columnCount。
  const innerRef = useRef<HTMLDivElement>(null);
  const [columnCount, setColumnCount] = useState(1);
  const [cardWidth, setCardWidth] = useState(172);

  useEffect(() => {
    if (!innerRef.current) return;
    const el = innerRef.current;

    const measure = () => {
      // 读 --card-w（已由 [data-density] 设置）。
      const cssCardW = parseFloat(
        getComputedStyle(el).getPropertyValue("--card-w").trim()
      );
      const w = Number.isFinite(cssCardW) && cssCardW > 0 ? cssCardW : 172;
      setCardWidth(w);

      // 内宽 = clientWidth - padding-left - padding-right。
      // px-8 = 2rem = 32px each side。
      const inner = el.clientWidth - 64;
      // auto-fill 等价 = floor((inner + COLUMN_GAP) / (cardW + COLUMN_GAP))
      const cols = Math.max(1, Math.floor((inner + COLUMN_GAP) / (w + COLUMN_GAP)));
      setColumnCount(cols);
    };

    measure();
    const ro = new ResizeObserver(measure);
    ro.observe(el);
    return () => ro.disconnect();
  }, []);

  const rowCount = Math.ceil(games.length / columnCount);
  const rowHeight = cardWidth * (4 / 3) + META_HEIGHT;

  const virtualizer = useVirtualizer({
    count: rowCount,
    getScrollElement: () => scrollContainerRef.current,
    estimateSize: () => rowHeight + ROW_GAP,
    overscan: 5,
  });

  // ── existing mutation callbacks（不动） ────────────────────────────────────
  const onRefreshCover = useCallback(
    async (game: Game) => {
      try {
        await refreshMetadata(game.id);
        if (onChildMutation) {
          onChildMutation();
        } else {
          const st = useLibraryStore.getState();
          const trimmed = st.searchQuery.trim();
          const queryArg = trimmed === "" ? null : trimmed;
          const filterArg =
            st.filter.tag_id == null &&
            st.filter.status == null &&
            !st.filter.favorite &&
            st.filter.brand == null &&
            st.filter.year_decade == null
              ? null
              : st.filter;
          try {
            const fresh = await searchGames(queryArg, st.sortBy, filterArg);
            setGames(fresh);
          } catch {
            const fresh = await listGames();
            setGames(fresh);
          }
        }
        toast.success("已刷新封面");
      } catch (e: unknown) {
        toast.error(`刷新封面失败 — ${String(e)}`);
      }
    },
    [setGames, onChildMutation],
  );

  const onChildMutated = useCallback(async () => {
    if (onChildMutation) {
      onChildMutation();
      return;
    }
    try {
      const fresh = await listGames();
      setGames(fresh);
    } catch (e: unknown) {
      console.error("[GameGrid] post-mutation refetch failed:", e);
    }
  }, [onChildMutation, setGames]);

  // ── render ────────────────────────────────────────────────────────────────
  // 注意：本组件不套 overflow-auto；scrollContainer 在 Library.tsx 第 258 行外层。
  return (
    <div ref={innerRef} className="px-8 pb-20 pt-7">
      <div
        style={{
          height: `${virtualizer.getTotalSize()}px`,
          width: "100%",
          position: "relative",
        }}
      >
        {virtualizer.getVirtualItems().map((row) => {
          const start = row.index * columnCount;
          const end = Math.min(start + columnCount, games.length);
          const rowGames = games.slice(start, end);
          return (
            <div
              key={row.key}
              data-row-index={row.index}
              style={{
                position: "absolute",
                top: 0,
                left: 0,
                width: "100%",
                transform: `translateY(${row.start}px)`,
                display: "grid",
                gridTemplateColumns: `repeat(${columnCount}, minmax(0, 1fr))`,
                columnGap: `${COLUMN_GAP}px`,
                rowGap: `${ROW_GAP}px`,
                alignItems: "start",
              }}
            >
              {rowGames.map((g) => (
                <GameCard
                  key={g.id}
                  game={g}
                  coverDataUrl={resolveCover(g)}
                  onPickMetadata={onPickMetadata}
                  onRefreshCover={onRefreshCover}
                  onMutated={onChildMutated}
                />
              ))}
            </div>
          );
        })}
      </div>
    </div>
  );
}
```
